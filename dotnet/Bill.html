             using System;
namespace Ash
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Enter Name: ");
            string Name = Console.ReadLine();
            Console.Write("Enter Id: ");
            string id = Console.ReadLine();
            Console.Write("Enter Unit: ");
            Int32 unit = Int32.Parse(Console.ReadLine());
            float charge, amount, surcharge=0;
            if (unit < 199)
            {
                charge = 1.20f;
            }
            else if(unit < 400)
            {
                charge = 1.50f;
            }
            else if(unit < 600)
            {
                charge = 1.80f;
            }
            else
            {
                charge = 2.0f;
            }
            amount = charge * unit;
            Console.WriteLine("\tCustomer ID\t\t\t\t{0}\n\tName\t\t\t\t{1}\n\tUnit Used\t\t\t\t{2}\n\tCharge @Rs {3} per Unit", id, Name.Trim(), unit, charge);
            Console.Write("\t Amount\t\t\t\t{0} \n", amount);
            if (amount > 400)
            {
                surcharge = amount * 0.15f;
                amount += surcharge;
            }
            
            if (surcharge > 0)
            {
                Console.WriteLine("\tSurchare Amount\t\t\t\t{0}", surcharge);
            }
            Console.WriteLine("--------------------------------------------\n\tAmount To Be Paid \t\t{0}", amount);
        }
    }
}
