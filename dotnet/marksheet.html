            using System;

namespace Programe
{
    class MyPrograme
    {


        static void Main() => MarkSheet("Dault",new string[]{"Math","Science", "English","Hindi"},new int[]{20,36,54,65});
            	
        static void MarkSheet(string name,string[] subjects,int[] number)
        {
            Console.WriteLine("\tName: {0}\n\n",name);
            float total = 0;
            for (int x = 0; x < subjects.Length; x++)
            {
                total += number[x];
                Console.WriteLine("{0}\t\t\t{1}\n",subjects[x],number[x]);
            }
            Console.WriteLine("\tTotal Precentage    {0}%",total/number.Length);
        }
    }
}
