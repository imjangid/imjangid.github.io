            using System;

namespace Programe
{
    class MyPrograme
    {	
    	static int[] Swap(int[] X)
        {
            int y = X[0];
            X[0] = X[1];
            X[1] = y; 
            return X;
        }

        static void Main()
        {
            int[] X = new int[]{1, 2};
            X = Swap(X);
            Console.WriteLine("\n\nSwappped  {0}  {1}",X[0],X[1]);
        }
     }
