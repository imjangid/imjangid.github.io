            using System;
namespace Ash
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Enter Month number: ");
            Char word = Console.ReadKey().KeyChar;
            switch (word)
            {
                case 'a':
                case 'e':
                case 'i':
                case 'o':
                case 'u':
                    Console.WriteLine("\nVowel");
                    break;
                default:
                    Console.WriteLine("\nConsonant");
                    break;

            }
           
        }
    }
}
