Temperature 
             using System;
using System.Linq;
namespace Ash
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.Write("Enter Temp: ");
            Int64 temp = Int64.Parse(Console.ReadLine());
            if (temp <0)
            {
                Console.WriteLine("Freezing wather");
            }
            else if (temp<10){
                Console.WriteLine("Very Cold wather");
            }
            else if (temp < 20)
            {
                Console.WriteLine("Cold wather");
            }
            else if (temp < 30)
            {
                Console.WriteLine("Normal in Temp");
            }
            else if (temp < 40)
            {
                Console.WriteLine("Its Hot");
            }
            else
            {
                Console.WriteLine("Very Hot");
            }
        }
    }
}
