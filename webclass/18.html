    $( function() {
    $( "#first" ).datepicker();

    } );
    function calc(){
    var a = $("#first").datepicker("getDate");
    var b = $("#second").datepicker("getDate");
    document.getElementById('output').innerHTML = Math.abs(a-b)/86400000;}
